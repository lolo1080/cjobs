var setJoinNetworkHref = null;

var init = function () {

	YAHOO.util.Event.onContentReady("navSummPanel", function() {
	
	
	
	
	
	
	
	

		loginPanel = new YAHOO.widget.Panel("loginPanelDiv", { width:"450px", visible:false, fixedcenter : true, constraintoviewport: false, modal : true, y: 25 });
		loginPanel.hide();
		loginPanel.render();
	});

	// Load up the buttons on the login panel.
	YAHOO.util.Event.onContentReady("loginPanelDiv", function() {
		var oLinkButton1 = new YAHOO.widget.Button("loginPanelButton");

	// Handle a login button click. Set up some fields for later use and show the login panel.
	function loginClicked(p_oEvent) {
		YAHOO.util.Event.stopEvent(p_oEvent);
		document.getElementById("destinationLink").value = "";
		document.getElementById("loginError").innerHTML = "";
		document.getElementById("loginPanelDiv").style.display = "block";
		loginPanel.show();
	}

	// Handle a join network button click. Set up some fields for later use and either call the processor or show the login panel.
	function joinThisNetworkClicked(joinNetworkHref) {
		if (isAuthenticated == "true") {
			document.location.href = joinNetworkHref;
		}
		else {
			submitAfterLogin = 4;
			document.getElementById("destinationLink").value = joinNetworkHref;

	}

	// Handle a registration button click on the login panel. Route the user to the original destination link

	function signUpNowClicked(p_oEvent) {
		// If a form name is provided then simply submit that form.
	
	
	
		// Call the appropriate processor based on which button was clicked to bring up the login dialog in the first place.
		document.location.href = document.getElementById("destinationLink").value + "&rts=1&ml=1";
	
	
	}

	// User clicked the login button on the panel. Send request to server for authentication.
	function logMeIn(p_oEvent) {
		var sjt_user_name = document.getElementById("SJT_USER_NAME").value;
		var sjt_password = document.getElementById("SJT_PASSWD").value;
		var xmlhttp = null;
		if (window.XMLHttpRequest) {
			// code for Firefox, Opera, IE7, etc.
			xmlhttp = new XMLHttpRequest();
		}
		else if (window.ActiveXObject) {
			// code for IE6, IE5
			xmlhttp = new ActiveXObject("Microsoft.XMLHTTP");
		}
		if (xmlhttp != null) {
			// Generate random to prevent caching.
			var d = new Date();
			var bigNumber = " " + d.getYear() + d.getDay() + d.getHours() + d.getMinutes() + d.getSeconds() + d.getMilliseconds();
			var randomNumber = (Math.floor(Math.random() * bigNumber));

			xmlhttp.onreadystatechange = logMeInResponded(xmlhttp, processLogMeInResponse);
			// Note: The dockey may not be populated.
			xmlhttp.open("GET", "http://seeker.dice.com/profman/servlet/ProfMan?op=2000&ajax=1&SUBMIT=1&SJT_USER_NAME=" + sjt_user_name + "&SJT_PASSWD=" + sjt_password + "&dockey= &random=" + randomNumber, true);
			xmlhttp.send(null);
		}
		else {
			//alert("Your browser does not support XMLHTTP.");
		}
	}

	// Server responded to out request for authentication. Check status codes.
	function logMeInResponded(xmlhttp, responseXmlHandler) {
		return function () {
			if (xmlhttp.readyState == 4) {
				// 4 = "loaded"
				if (xmlhttp.status == 200) {
					// 200 = "OK"
					//alert("xml: " + xmlhttp.responseText);
					responseXmlHandler(xmlhttp.responseXML);
				}
				else {
					//alert("Problem retrieving data:" + xmlhttp.statusText);
				}
			}
		}
	}

	// Process the server's response to our request for authentication.
	function processLogMeInResponse(locations) {
		var locationsVals = locations.getElementsByTagName("locations")[0];
		// If we didn't get anything back from the server,
		// exit here or risk the wrath of IE's stellar javascript error detection.
		if (locationsVals == null) {
			return;
		}
		var items = locationsVals.getElementsByTagName("item");

		for (var I = 0; I < items.length; I++) {
			var item = items[I];
			validLogin = item.getElementsByTagName("validLogin")[0].firstChild.nodeValue;

			if (validLogin == 1) {
				//var loginPanel = new YAHOO.widget.Panel("loginPanelDiv").hide();

				document.body.className = "yui-skin-sam";

				var jobRating;
				var applicationDate;
				var jobInfoPresent = item.getElementsByTagName("jobInfoPresent")[0].firstChild.nodeValue;
				if (jobInfoPresent == "1") {
					if (item.getElementsByTagName("jobRating")[0].firstChild != null) {
						jobRating = item.getElementsByTagName("jobRating")[0].firstChild.nodeValue;
					}
					if (item.getElementsByTagName("applicationDate")[0].firstChild != null) {
						applicationDate = item.getElementsByTagName("applicationDate")[0].firstChild.nodeValue;
					}
					if (item.getElementsByTagName("jobReportedDate")[0].firstChild != null) {
						jobReportedDate = item.getElementsByTagName("jobReportedDate")[0].firstChild.nodeValue;
					}
				}

			
			
			
				if (submitAfterLogin == 1) {
					// Call the appropriate processor based on which button was clicked to bring up the login dialog in the first place.
					document.location.href = document.getElementById("destinationLink").value;
				}
				else {
					// We're staying on this page. Update any information that needs to be updated.
					loginPanel.hide();
					isAuthenticated = "true";

					// Rate This Job was clicked. Populate the rate job dropdown with seeker's rating information for
					// this job before showing the rating dropdown.
					if (submitAfterLogin == 2) {
						rateJobPanelShow();
					}
					else if (submitAfterLogin == 3) {
						// Report This Job was clicked. Determine which dropdown to show based on whether this seeker has
						// previously reported this job.
						if (jobReportedDate == null || jobReportedDate == "") {
							reportJobPanelShow();
						}
						else {
							reportJobPanelReportedShow();
						}
					} else if (submitAfterLogin == 4) {
						// Join Network was clicked.
						document.location.href = document.getElementById("destinationLink").value;
					}
				}
			
			
			}
			else {
				// AUthorization failed. Inform the user.
				var error = item.getElementsByTagName("error")[0].firstChild.nodeValue;
				var loginError = document.getElementById("loginError");
				loginError.innerHTML = error.replace("\n", "<br>");
			}
		}
	}
}();

function toggleNavigationState(id){
	if (document.NAVSUMMARY.EXPANDED_NE.value == '') {
		expandNavigation(id);
	} else if (document.NAVSUMMARY.EXPANDED_NE.value == id) {
		collapseNavigation(id);
	} else {
		collapseNavigation(document.NAVSUMMARY.EXPANDED_NE.value);
		expandNavigation(id);
	}
	return true;
}
